// Get list element
const LIST_ELEMENT = document.getElementById("list");
            
// A function that returns a new list item element.
function createListItem(innerText) {
    const item = document.createElement("li");
    const text = document.createTextNode(innerText);
    item.appendChild(text);
    return item;
}
            
// A function used to add names to the list
function addName(inputID) {
    const nameInput = document.getElementById(inputID);
            
    if (nameInput.value == "") {
        alert("Name cannot be empty!");
        return;
    }
            
    LIST_ELEMENT.appendChild(
        createListItem(nameInput.value)
    );
            
    nameInput.value = "";
}
            
function removeFirst() {
    LIST_ELEMENT.firstElementChild?.remove();
}
            
function removeLast() {
    LIST_ELEMENT.lastElementChild?.remove();
}
